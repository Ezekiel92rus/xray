.packet_loop:
            cmp  byte [esi-2],0xC
            jne  .data_packet
            cmp  dword[esi],0x12071980
            jne  .data_packet
            cmp  dword[esi+4],0x26111975

            cmp  byte [esi-2],0x0C

      movzx  eax,word[esi-2]

      cmp    edi,esi

      xor  eax,eax
      jmp  .packet_ret

      mov eax,1
.packet_ret:
      pop	edx ecx ebx edi esi
      ret
